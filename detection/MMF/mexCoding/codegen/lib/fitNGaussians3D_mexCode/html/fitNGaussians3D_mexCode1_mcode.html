1function [F,J] = fitNGaussians3D_mexCode(x0,image,index,psfSigma) %#codegen
  2% Edit of fitNGaussians2D to work in 3D
  3%   EHarry March 2012
  4
  5%% ORIGINAL HEADER
  6% % %FITNGAUSSIANS2D yields F, the difference between an image and a theoretical image produced by N Gaussians, and J, the Jacobian of F.
  7% % %
  8% % %SYNOPSIS [F,J] = fitNGaussians2D(x0,image,index,psfSigma)
  9% % %
 10% % %INPUT  x0      : initial guess of PSF positions and amplitudes and
 11% % %                 background noise.
 12% % %       image   : Image part being analyzed.
 13% % %       index   : x,y-indices of pixels considered.
 14% % %       psfSigma: Standard deviation of point spread function (in pixels).
 15% %
 16% % %OUTPUT F       : Residuals from fitting an image with supplied
 17% % %                 Gaussians.
 18% % %       J       : The Jacobian matrix of F.
 19% % %       errFlag : 0 if function executes normally, 1 otherwise.
 20% % %
 21% % %REMARKS F = model image - real image, important to know if the sign of the
 22% % %residuals matters.
 23% % %
 24% % %Khuloud Jaqaman, August 2005
 25
 26%% Output
 27
 28F = [];
 29J = [];
 30
 31%% Input
 32
 33% %check whether correct number of input arguments was used
 34% if nargin ~= 4
 35%     disp('--fitNGaussians2D: Incorrect number of input arguments!');
 36%     return
 37% end
 38%check whether correct number of input arguments was used
 39if nargin ~= 4
 40    disp('--fitNGaussians3D: Incorrect number of input arguments!');
 41    return
 42end
 43
 44%% Calculating F & J
 45
 46%extract background intensity from x0 and remove from vector
 47bgAmp = x0(end);
 48x0 = x0(1:end-1);
 49
 50%get number of PSFs considered
 51% numPSF = length(x0)/3;
 52numPSF = length(x0)/4;
 53
 54% %reshape 3nx1 vector x0 into nx3 matrix
 55% x0 = reshape(x0,3,numPSF);
 56% x0 = x0';
 57
 58%reshape 4nx1 vector x0 into nx4 matrix
 59x0 = reshape(x0,4,numPSF);
 60x0 = x0';
 61
 62%extract PSF center positions and amplitudes
 63% psfPos = x0(:,1:2);
 64% psfAmp = x0(:,3);
 65
 66psfPos = x0(:,1:3);
 67psfAmp = x0(:,4);
 68
 69%find minimum and maximum pixel indices
 70minIndxX = min(index(:,1));
 71maxIndxX = max(index(:,1));
 72minIndxY = min(index(:,2));
 73maxIndxY = max(index(:,2));
 74minIndxZ = min(index(:,3));
 75maxIndxZ = max(index(:,3));
 76
 77%determine the contribution of each PSF (assuming amplitude 1) to a
 78%pixel based on its x-coordinate (needed to calculate F & J)
 79psfIntegX = zeros(maxIndxX-minIndxX+1,numPSF);
 80for i=1:numPSF
 81    temp = GaussListND_mexCode((minIndxX:maxIndxX)',...
 82        psfSigma(1),psfPos(i,1));
 83    
 84    temp2 = squeeze(temp);
 85    %clear temp
 86    
 87    psfIntegX(:,i) = temp2(:,1);
 88    %clear temp2
 89end
 90
 91%determine the contribution of each PSF (assuming amplitude 1) to a
 92%pixel based on its y-coordinate (needed to calculate F & J)
 93psfIntegY = zeros(maxIndxY-minIndxY+1,numPSF);
 94for i=1:numPSF
 95    temp = GaussListND_mexCode((minIndxY:maxIndxY)',...
 96        psfSigma(1),psfPos(i,2));
 97    
 98    temp2 = squeeze(temp);
 99    %clear temp
100    
101    psfIntegY(:,i) = temp2(:,1);
102    %clear temp2
103end
104
105%determine the contribution of each PSF (assuming amplitude 1) to a
106%pixel based on its z-coordinate (needed to calculate F & J)
107psfIntegZ = zeros(maxIndxZ-minIndxZ+1,numPSF);
108for i=1:numPSF
109    temp = GaussListND_mexCode((minIndxZ:maxIndxZ)',...
110        psfSigma(2),psfPos(i,3));
111    
112    
113    temp2 = squeeze(temp);
114    %clear temp
115    
116    psfIntegZ(:,i) = temp2(:,1);
117    %clear temp2
118end
119
120%calculate the value of each PSF (assuming amplitude 1) at the
121%x-coordinates of the corners of all pixels (needed to calculate J)
122psfValueX = zeros(maxIndxX-minIndxX+2,numPSF);
123for i=1:numPSF
124    psfValueX(:,i) = exp(-((minIndxX-0.5:maxIndxX+0.5)'...
125        -psfPos(i,1)).^2/2/psfSigma(1)^2);
126end
127
128%calculate the value of each PSF (assuming amplitude 1) at the
129%y-coordinates of the corners of all pixels (needed to calculate J)
130psfValueY = zeros(maxIndxY-minIndxY+2,numPSF);
131for i=1:numPSF
132    psfValueY(:,i) = exp(-((minIndxY-0.5:maxIndxY+0.5)'...
133        -psfPos(i,2)).^2/2/psfSigma(1)^2);
134end
135
136%calculate the value of each PSF (assuming amplitude 1) at the
137%z-coordinates of the corners of all pixels (needed to calculate J)
138psfValueZ = zeros(maxIndxZ-minIndxZ+2,numPSF);
139for i=1:numPSF
140    psfValueZ(:,i) = exp(-((minIndxZ-0.5:maxIndxZ+0.5)'...
141        -psfPos(i,3)).^2/2/psfSigma(2)^2);
142end
143
144%get number of pixels in image
145numPixel = length(image);
146
147% %get xy-indices relative to minimum
148% relIndxX = index(:,1) - minIndxX + 1;
149% relIndxY = index(:,2) - minIndxY + 1;
150%get xy-indices relative to minimum
151relIndxX = index(:,1) - minIndxX + 1;
152relIndxY = index(:,2) - minIndxY + 1;
153relIndxZ = index(:,3) - minIndxZ + 1;
154
155% %calculate the value of F at all pixels
156% F = (sum(repmat(psfAmp,1,numPixel).*psfIntegX(relIndxX,:)'.*psfIntegY(relIndxY,:)',1))' ...
157%     + repmat(bgAmp,numPixel,1) - image;
158
159%calculate the value of F at all pixels
160F = (sum(repmat(psfAmp,1,numPixel).*psfIntegX(relIndxX,:)'.*psfIntegY(relIndxY,:)'.*psfIntegZ(relIndxZ,:)',1))' ...
161    + repmat(bgAmp,numPixel,1) - image;
162
163%remove pixels with NaN (which means they are out of the cropped image
164%area)
165indxPixel = find(~isnan(image));
166
167F = F(indxPixel);
168
169if nargout > 1
170    %calculate the derivative at all pixels
171    %     J = ones(numPixel,3*numPSF+1); %(last column for background amplitude)
172    %     J(:,1:3:3*numPSF) = repmat(psfAmp',numPixel,1).*(psfValueX(relIndxX,:)-...
173    %         psfValueX(relIndxX+1,:)).*psfIntegY(relIndxY,:); %w.r.t. x
174    %     J(:,2:3:3*numPSF) = repmat(psfAmp',numPixel,1).*(psfValueY(relIndxY,:)-...
175    %         psfValueY(relIndxY+1,:)).*psfIntegX(relIndxX,:); %w.r.t. y
176    %     J(:,3:3:3*numPSF) = psfIntegX(relIndxX,:).*psfIntegY(relIndxY,:); %w.r.t. amp
177    
178    J = ones(numPixel,4*numPSF+1); %(last column for background amplitude)
179    J(:,1:4:4*numPSF) = repmat(psfAmp',numPixel,1).*(psfValueX(relIndxX,:)-...
180        psfValueX(relIndxX+1,:)).*psfIntegY(relIndxY,:).*psfIntegZ(relIndxZ,:); %w.r.t. x
181    J(:,2:4:4*numPSF) = repmat(psfAmp',numPixel,1).*(psfValueY(relIndxY,:)-...
182        psfValueY(relIndxY+1,:)).*psfIntegX(relIndxX,:).*psfIntegZ(relIndxZ,:); %w.r.t. y
183    J(:,3:4:4*numPSF) = repmat(psfAmp',numPixel,1).*(psfValueZ(relIndxZ,:)-...
184        psfValueZ(relIndxZ+1,:)).*psfIntegX(relIndxX,:).*psfIntegY(relIndxY,:); %w.r.t. z
185    J(:,4:4:4*numPSF) = psfIntegX(relIndxX,:).*psfIntegY(relIndxY,:).*psfIntegZ(relIndxZ,:); %w.r.t. amp
186    
187    %remove pixels with NaN (which means they are out of the cropped image
188    %area)
189    J = J(indxPixel,:);
190end
191
192%% ~~ the end ~~
