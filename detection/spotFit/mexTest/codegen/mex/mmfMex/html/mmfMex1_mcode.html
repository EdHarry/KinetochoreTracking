1function [ jacobian,resi,amp,bg,nGauss,nDim ] = mmfMex( maskAmp,maskCoord,X,sigma )
  2%MMFMEX Summary of this function goes here
  3%   Detailed explanation goes here
  4
  5%% INITIAL ALLOCATION
  6%To get bg from fitting, fit [G|1] * (a,bg)'
  7%Ones -> To obtain the bg column from the start. gaussList is 1-n
  8%gauss + one col of ones for bg.
  9%gaussList = ones(size(maskAmp,1),size(X,1)+1);
 10%Number of gaussian is equal to the number of spots;
 11%nGauss = nSpot;
 12nGauss = size(X,1);
 13nDim = size(X,2);
 14
 15%gaussList = GaussListND_mexCode_(maskCoord,sigma,X);
 16
 17%nGauss = size(centerInput,1);
 18nCoords = size(maskCoord,1);
 19
 20gaussListOutput = ones(nCoords,nGauss+1);
 21
 22sigma2 = repmat(sigma,[nCoords,1]);
 23
 24tmp = 0.5./sigma2;
 25
 26normN = (2*pi)^(0.5*nDim)*prod(sigma2(1,:));
 27
 28sq2 = sqrt(2);
 29
 30for gaussIdx = 1:nGauss
 31    
 32    
 33    
 34    
 35    center2 = repmat(X(gaussIdx,:), [nCoords,1]);
 36    
 37    
 38    
 39    %======================
 40    % CALC GAUSSLIST
 41    %======================
 42    
 43    
 44    % instead of calculating Gauss-values for very complicated geometries, we
 45    % make a coordinate transformation so that we can use sigma=1 in all
 46    % dimensions
 47    
 48    
 49    
 50    % 0.5*erfc(-(x+0.5)/sqrt(2))-0.5*erfc(-(x-0.5)/sqrt(2)) gives the integral on the
 51    % pixel at 1 of a Gaussian with mean 0 and sigma 1
 52    
 53    
 54    %center3 = center2(:,1);
 55    %clear center2
 56    
 57    % convert coordList to 0/1
 58    %coordList2 = (coordList(1:nCoords,1:nDims) - center2(1:nCoords,1:nDims))./sigma2(1:nCoords,1:nDims);
 59    coordList2 = (maskCoord- center2)./sigma2;
 60    %clear coordList center3
 61    
 62    % double coordList as preparation for erfc
 63    %fixed bug: must divide the 0.5 by sigma - KJ
 64    %coordList2 = cat(3,coordList2-0.5./sigma2(1:nCoords,1:nDims), coordList2+0.5./sigma2(1:nCoords,1:nDims));
 65    
 66    coordList2 = cat(3,coordList2-tmp, coordList2+tmp);
 67    
 68    % calculate gaussList
 69    %Jonas was missing the minus sign in erfc. I corrected that - KJ
 70    gaussList = diff(0.5 * erfc(-coordList2/sq2),1,3);
 71    gaussList = prod(gaussList,2);
 72    
 73    % norm gaussList
 74    gaussListOutput(:,gaussIdx) = gaussList(:,1) * normN;
 75    
 76end
 77
 78
 79%% GAUSS LIST CALCULATION
 80% for gaussIdx = 1:nGauss
 81%     tmp = GaussListND_mexCode(maskCoord,sigma,X(gaussIdx,:));
 82%     gaussList(:,gaussIdx) = tmp(:,1);
 83%     %gaussList(:,gaussIdx) = GaussListND_mexCode_mex(maskCoord,sigma,X(gaussIdx,:));
 84% end
 85
 86
 87%% GAUSS FITTING
 88%I = (G1,G2,G3,....,1) *(a1;a2;a3;...;bg);
 89%To fit -> a1;a2..;bg = (G1,G2,G3,...,1) \ I
 90newAmpBg = gaussListOutput \ maskAmp;
 91
 92%% RESIDUAL CALCULATION
 93%Resi : I - I' -> I' = (G1,G2,G3,G4...1) * (a1;a2;a3;...;bg);
 94resi = maskAmp - ( gaussListOutput * newAmpBg );
 95bg = newAmpBg(end);
 96amp = newAmpBg(1:end-1);
 97
 98
 99sigmaSq = (sigma.^2);
100
101%% ASSEMBLY of JACOBIAN MATRIX
102%Compute gradiant of function --> aG(x,y,z)+bg = I;
103%Preallocation : jacobian is
104%[x1,y1,z1,x2,y2,z2...,xn,yn,zn,a1,a2..,an,bg];
105%jacobian = zeros(size(gaussList,1), ( nGauss*nDim + nGauss + 1) );
106%Increased gaussList
107jacobian = repeatEntries_mex(gaussListOutput(:,1:end-1)',nDim)';
108%dI/da = G and dI/dbg = 1
109jacobian = cat(2,jacobian,gaussListOutput);
110%For each coordinate, calculate the first partial derivative--> for
111%x,y,z is a*g*(x-c)^2/2*sigma^2 * a*-(x-c)/sigma^2
112for cJacobian = 1:(nGauss)
113    col = (cJacobian -1) * nDim + 1;%x..y..z
114    upperCol = col+nDim-1;
115    if(amp(cJacobian) < 0)
116        %jacobian(:,col:upperCol) = bsxfun(@times,jacobian(:,col:upperCol),-amp(cJacobian));
117        jacobian(:,col:upperCol) = jacobian(:,col:upperCol) * -amp(cJacobian);
118    else
119        %jacobian(:,col:upperCol) = bsxfun(@times,jacobian(:,col:upperCol),amp(cJacobian));
120        jacobian(:,col:upperCol) = jacobian(:,col:upperCol) * amp(cJacobian);
121    end
122    jacobian(:,col:upperCol) = -jacobian(:,col:upperCol) .* bsxfun(@rdivide,bsxfun(@minus,maskCoord,X(cJacobian,:)),sigmaSq);
123    %tmp = -jacobian(:,col:upperCol) .* ((maskCoord - X(cJacobian,:)) ./ sigmaSq);
124    %jacobian(:,col:upperCol) = tmp(:,1);
125end

127
128
129end
